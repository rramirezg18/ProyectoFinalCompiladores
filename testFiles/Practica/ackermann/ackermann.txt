programx{
    main(){

        int m = 2;
        int n = 3;

        int resultado = ackermann(m, n);
        print(resultado);
    }

    int ackermann(int m, int n){
        if (m == 0) {
            return n + 1;
        }
        else if (n == 0) {
            return ackermann(m - 1, 1);
        }
        else {
            return ackermann(m - 1, ackermann(m, n - 1));
        }
    }
}
